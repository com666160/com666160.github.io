<script>
    // ========== 核心：使用 JSONBin.io 作为免费后端存储（无需服务器）==========
    // 这是免费的 JSON 存储 API，可以保存和读取分享的代码
    // 您也可以换成自己的后端，这里用公开服务演示
    
    // 注意：为了让大家都能用，这里使用公共的 JSONBin 示例
    // 实际使用时建议注册免费账号获取自己的 API Key（完全免费）
    const JSONBIN_BIN_ID = "679d9f8ae41b4d34e480c1a5";  // 示例 bin ID（演示用）
    const JSONBIN_API_KEY = "$2a$10$QrZzZzZzZzZzZzZzZzZzZu";  // 占位符，实际需要替换
    
    // 更好的方案：使用 LocalStorage + URL Hash 的混合方案（无需后端也能分享）
    // 为了让系统立即可用，我实现两种模式：
    // 模式1：基于 URL Hash（无需后端，复制链接即可分享）
    // 模式2：点击发布生成短链接（使用免费的 ShortLink 服务）
    
    const textarea = document.getElementById('htmlCode');
    const previewFrame = document.getElementById('previewFrame');
    const runBtn = document.getElementById('runBtn');
    const publishBtn = document.getElementById('publishBtn');
    const copyLinkBtn = document.getElementById('copyLinkBtn');
    const demoBtn = document.getElementById('demoBtn');
    const shareLinkInput = document.getElementById('shareLinkInput');
    
    // 更新预览
    function updatePreview() {
        let code = textarea.value;
        if (!code.trim()) {
            code = '<!DOCTYPE html><html><head><meta charset="UTF-8"><title>空白页面</title></head><body style="display:flex;align-items:center;justify-content:center;height:100vh;font-family:sans-serif;"><h2>✨ 欢迎使用</h2><p>在左侧编写 HTML 代码，点击「生成在线链接」分享给任何人</p></body></html>';
            textarea.value = code;
        }
        const iframeDoc = previewFrame.contentDocument || previewFrame.contentWindow.document;
        iframeDoc.open();
        iframeDoc.write(code);
        iframeDoc.close();
    }
    
    // 从 URL hash 加载代码
    function loadFromHash() {
        const hash = window.location.hash.substring(1);
        if (hash && hash.startsWith('code=')) {
            try {
                const decoded = decodeURIComponent(hash.substring(5));
                if (decoded) {
                    textarea.value = decoded;
                    updatePreview();
                    shareLinkInput.value = window.location.href;
                    showToast('✅ 已从链接加载代码', false);
                    return true;
                }
            } catch(e) {}
        }
        return false;
    }
    
    // 生成基于 Hash 的分享链接（立即可用，无需后端）
    function generateHashLink() {
        const code = textarea.value;
        if (!code.trim()) {
            showToast('请先编写一些代码', true);
            return null;
        }
        const encoded = encodeURIComponent(code);
        const baseUrl = window.location.href.split('#')[0];
        const newUrl = `${baseUrl}#code=${encoded}`;
        return newUrl;
    }
    
    // 显示提示
    function showToast(msg, isError) {
        let toast = document.querySelector('.toast');
        if (!toast) {
            toast = document.createElement('div');
            toast.className = 'toast';
            toast.style.cssText = 'position:fixed;bottom:20px;left:50%;transform:translateX(-50%);background:#1e293b;color:white;padding:10px 24px;border-radius:40px;z-index:9999;font-size:14px;';
            document.body.appendChild(toast);
        }
        toast.textContent = msg;
        toast.style.background = isError ? '#dc2626' : '#10b981';
        toast.style.opacity = '1';
        setTimeout(() => { toast.style.opacity = '0'; }, 3000);
    }
    
    // 发布：生成在线链接（使用免费的 TinyURL 或 直接显示 Hash 链接）
    async function publishAndGetLink() {
        const hashLink = generateHashLink();
        if (!hashLink) return null;
        
        // 方案A：直接使用 Hash 链接（最可靠，无需任何后端）
        // 这个链接已经是完整的在线链接，复制后发给任何人，对方打开就能看到代码！
        shareLinkInput.value = hashLink;
        
        // 可选：尝试使用免费短链接服务缩短 URL
        try {
            // 调用 TinyURL 的免费 API 缩短链接（可选，不影响核心功能）
            const response = await fetch(`https://tinyurl.com/api-create.php?url=${encodeURIComponent(hashLink)}`);
            if (response.ok) {
                const shortUrl = await response.text();
                shareLinkInput.value = shortUrl;
                showToast('✨ 已生成短链接，复制即可分享', false);
                return shortUrl;
            }
        } catch(e) {
            // 短链接服务失败，仍使用原始 Hash 链接
            console.log('短链接服务不可用，使用原始链接');
        }
        
        showToast('✅ 链接已生成！复制后发给任何人', false);
        return hashLink;
    }
    
    // 复制链接
    async function copyLink() {
        const link = shareLinkInput.value;
        if (!link || link === '点击「生成在线链接」获得可分享URL') {
            showToast('请先点击「生成在线链接」', true);
            return;
        }
        try {
            await navigator.clipboard.writeText(link);
            showToast('📋 链接已复制到剪贴板！', false);
        } catch(e) {
            showToast('手动复制链接吧', true);
        }
    }
    
    // 加载示例
    function loadDemo() {
        const demo = `<!DOCTYPE html>
<html lang="zh">

    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">

        .card {

        h1 { color: #1e1b4b; margin-bottom: 16px; }

    <div class="card">
        <h1>🎉 分享成功！</h1>
        <p>这个页面是通过「HTML工坊」创建的</p>
        <div class="counter" id="count">0</div>
        <button onclick="document.getElementById('count').innerText++">点我 +1</button>
        <button onclick="document.getElementById('count').innerText=0" style="background:#64748b;">重置</button>
        <p style="margin-top:24px; font-size:12px; color:#666;">✨ 任何人都能通过链接看到这个页面</p>
    </div>
</body>
</html>`;
        textarea.value = demo;
        updatePreview();
        showToast('已加载示例，点击「生成在线链接」即可分享', false);
    }
    
    // 监听 hash 变化（别人打开分享链接时自动加载）
    window.addEventListener('hashchange', () => {
        loadFromHash();
    });
    
    // 初始化
    function init() {
        // 绑定事件
        runBtn.addEventListener('click', updatePreview);
        publishBtn.addEventListener('click', async () => {
            const link = await publishAndGetLink();
            if (link) {
                // 同时更新地址栏
                if (!window.location.href.includes('#code=')) {
                    history.pushState(null, '', link.split('#')[0] + '#code=' + encodeURIComponent(textarea.value));
                }
            }
        });
        copyLinkBtn.addEventListener('click', copyLink);
        demoBtn.addEventListener('click', loadDemo);
        
        // 尝试从 hash 加载，否则加载默认内容
        const loaded = loadFromHash();
        if (!loaded) {
            loadDemo();
        }
        
        // 快捷键 Ctrl+Enter 刷新
        textarea.addEventListener('keydown', (e) => {
            if ((e.ctrlKey || e.metaKey) && e.key === 'Enter') {
                e.preventDefault();
                updatePreview();
            }
        });
        
        updatePreview();
    }
    
    init();
</script>
